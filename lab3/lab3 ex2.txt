import java.util.Scanner;

public class lab_03 {
	static Scanner in = new Scanner(System.in);
public static void main(String args[]) {
	
	System.out.println("Sirul de caractere: ");
	String forConversion = in.next();
	
	System.out.println("Caracterul de cautat: ");
	char search = in.next().charAt(0);
	
	char [] arrayOfChar = forConversion.toCharArray();

	int k=0;
	
	for(int i=0;i<arrayOfChar.length;i++){
		if(arrayOfChar[i]==search)
			k++;
	}
	System.out.println("Nr. de aparitii: " +k);
	
}
}
